namespace = united_nations

# THIS leaves the UNGA and UNSC when THIS becomes a puppet
united_nations.1 = {
	type = country_event

	title = united_nations.1.t
	desc = united_nations.1.d
	
	#flavor = united_nations.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	duration = 1

	trigger = {
	
		OR = { 
		
			#THIS is UN recognized BUT has unrecognized country type
			AND = {
				OR = { 
					has_modifier = unga_member
					has_modifier = unsc_permanent_member
					has_modifier = unsc_member
				}
				NOT = { is_country_type = recognized }
			}
			
			#THIS is not UN recognized BUT has recognized country type
			AND = {
				NOT = { 
					has_modifier = unga_member
				}
				is_country_type = recognized
			}
			
			#THIS is UN recognized or has recognized country type but is a subject
			AND = {
				OR = { 
					has_modifier = unga_member
					has_modifier = unsc_permanent_member
					has_modifier = unsc_member
					is_country_type = recognized
				}
				is_sufficiently_independent = no
			}
			
		}
		
	}

	immediate = {
	
		#Clean up if wrongly unrecognized
		if = {
			limit = {
				OR = { 
					has_modifier = unga_member
					has_modifier = unsc_permanent_member
					has_modifier = unsc_member
				}
				NOT = { is_country_type = recognized }
			}
			set_country_type = recognized
		}
		
		#Clean up wrongly recognized (because THIS is recognized when THIS is not a member of the UNGA or UNSC)
		if = {
			limit = {
				NOT = { 
					has_modifier = unga_member
				}
				is_country_type = recognized
			}
			set_country_type = unrecognized
		}
		
		#Clean up wrongly recognized (because THIS is a subject)
		if = {
			limit = {
				OR = { 
					has_modifier = unga_member
					has_modifier = unsc_permanent_member
					has_modifier = unsc_member
					is_country_type = recognized
				}
				is_sufficiently_independent = no
			}
			remove_modifier = unga_member
			remove_modifier = unsc_permanent_member
			remove_modifier = unsc_member
			set_country_type = unrecognized
		}
		
	}

	option = {
		name = united_nations.1.a #Unfortunate
	}
	
}
