grant_ministership_to_politician = {
	icon = "gfx/interface/icons/generic_icons/in_government_icon.dds"
	clicksound = "event:/SFX/DLC/1.3_ip1/UI/agitator_promote"

	potential = {
		has_role = politician
		NOT = { has_modifier = THIS_is_ruler }
		NOT = { has_modifier = cabinet_minister }
		owner = scope:actor
	}

	possible = {
	
		interest_group = {
			is_in_government = yes
		}
		
	}

	effect = {
		add_modifier = { name = cabinet_minister }
		this.interest_group = {
			add_modifier = { name = cabinet_minister_ig }
		}
	}

	cooldown = { months = 1 }

	ai_chance = {
		value = -50
		if = {
			limit = {
				popularity > interest_group.leader.popularity
			}
			add = 25
		}
		if = {
			limit = {
				interest_group.leader.popularity <= -50
			}
			add = 50
		}
		if = {
			limit = {
				interest_group.leader.popularity >= 50
			}
			add = -50
		}
		if = {
			limit = {
				interest_group = {
					ig_approval <= 0
				}
			}
			add = 25
		}
		if = {
			limit = {
				this.ideology = owner.ruler.ideology
			}
			add = 50
		}
	}
}

remove_ministership_from_politician = {
	icon = "gfx/interface/icons/generic_icons/in_opposition_icon.dds"
	clicksound = "event:/SFX/DLC/1.3_ip1/UI/agitator_promote"

	potential = {
		has_role = politician
		has_modifier = cabinet_minister
		owner = scope:actor
	}

	possible = {
	
		interest_group = {
			is_in_government = yes
		}
		
	}

	effect = {
		remove_modifier = cabinet_minister
		this.interest_group = {
			remove_modifier = cabinet_minister_ig
			hidden_effect = {
				remove_modifier = ig_angry_with_government
				remove_modifier = THIS_is_ruler
			}
			add_modifier = { name = ig_angry_with_government years = 4 is_decaying = yes }
		}
	}

	cooldown = { months = 1 }

	ai_chance = {
		value = -50
		if = {
			limit = {
				popularity > interest_group.leader.popularity
			}
			add = 25
		}
		if = {
			limit = {
				interest_group.leader.popularity <= -50
			}
			add = 50
		}
		if = {
			limit = {
				interest_group.leader.popularity >= 50
			}
			add = -50
		}
		if = {
			limit = {
				interest_group = {
					ig_approval <= 0
				}
			}
			add = 25
		}
		if = {
			limit = {
				this.ideology = owner.ruler.ideology
			}
			add = 50
		}
	}
}
