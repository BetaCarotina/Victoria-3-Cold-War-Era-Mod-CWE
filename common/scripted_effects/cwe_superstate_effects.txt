count_superstate_members = {

	#Determine members
	set_variable = { 
		name = superstate_members
		value = 0
	}
	every_country = {
		limit = { 
			OR = { 
				AND = { has_variable = XEU_ambitions ROOT = { has_variable = XEU_ambitions } }
				AND = { has_variable = XSA_ambitions ROOT = { has_variable = XSA_ambitions } }
			}
		}
		ROOT = {
			change_variable = { 
				name = superstate_members
				add = 1
			}
		}
	}

}
