Asian_Development_Bank = {
	type = country
	first_valid = yes
	
	#Borrower
	text = {
		trigger = {
			
			any_country = {
				c:JAP ?= this
				has_diplomatic_pact = { who = ROOT type = multilateral_development_bank is_initiator = yes }
			}
			
		}
		localization_key = Asian_Development_Bank_borrower
	}
	
	#Creditor
	text = {
		trigger = {
		
			c:JAP ?= this
		
			any_country = {
				has_diplomatic_pact = { who = ROOT type = multilateral_development_bank is_initiator = no }
			}
			
		}
		localization_key = Asian_Development_Bank_creditor
	}
	
}
